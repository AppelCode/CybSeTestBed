12 def _is_within_distance_ahead(target_location, current_location, orientation, max_distance):
13  """
14  Check if a target object is within a certain distance in front of a reference object.
15 
16  :param target_location: location of the target object
17  :param current_location: location of the reference object
18  :param orientation: orientation of the reference object
19  :param max_distance: maximum allowed distance
20  :return: True if target object is within max_distance ahead of the reference object
21  """
22  target_vector = np.array([target_location.x - current_location.x, target_location.y - current_location.y])
23  norm_target = np.linalg.norm(target_vector)
24 
25  # If the vector is too short, we can simply stop here
26  if norm_target < 0.001:
27  return True
28 
29  if norm_target > max_distance:
30  return False
31 
32  forward_vector = np.array(
33  [math.cos(math.radians(orientation)), math.sin(math.radians(orientation))])
34  d_angle = math.degrees(math.acos(np.dot(forward_vector, target_vector) / norm_target))
35 
36  return d_angle < 90.0
