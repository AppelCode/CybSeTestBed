78  def _is_light_red_europe_style(self, lights_list):
79  """
80  This method is specialized to check European style traffic lights.
81 
82  :param lights_list: list containing TrafficLight objects
83  :return: a tuple given by (bool_flag, traffic_light), where
84  - bool_flag is True if there is a traffic light in RED
85  affecting us and False otherwise
86  - traffic_light is the object itself or None if there is no
87  red traffic light affecting us
88  """
89  ego_vehicle_location = self._vehicle.get_location()
90  ego_vehicle_waypoint = self._map.get_waypoint(ego_vehicle_location)
91 
92  for traffic_light in lights_list:
93  object_waypoint = self._map.get_waypoint(traffic_light.get_location())
94  if object_waypoint.road_id != ego_vehicle_waypoint.road_id or \
95  object_waypoint.lane_id != ego_vehicle_waypoint.lane_id:
96  continue
97 
98  loc = traffic_light.get_location()
99  if is_within_distance_ahead(loc, ego_vehicle_location,
100  self._vehicle.get_transform().rotation.yaw,
101  self._proximity_threshold):
102  if traffic_light.state == carla.TrafficLightState.Red:
103  return (True, traffic_light)
104 
105  return (False, None)

107  def _is_light_red_us_style(self, lights_list, debug=False):
108  """
109  This method is specialized to check US style traffic lights.
110 
111  :param lights_list: list containing TrafficLight objects
112  :return: a tuple given by (bool_flag, traffic_light), where
113  - bool_flag is True if there is a traffic light in RED
114  affecting us and False otherwise
115  - traffic_light is the object itself or None if there is no
116  red traffic light affecting us
117  """
118  ego_vehicle_location = self._vehicle.get_location()
119  ego_vehicle_waypoint = self._map.get_waypoint(ego_vehicle_location)
120 
121  if ego_vehicle_waypoint.is_intersection:
122  # It is too late. Do not block the intersection! Keep going!
123  return (False, None)
124 
125  if self._local_planner.target_waypoint is not None:
126  if self._local_planner.target_waypoint.is_intersection:
127  min_angle = 180.0
128  sel_magnitude = 0.0
129  sel_traffic_light = None

131  loc = traffic_light.get_location()
132  magnitude, angle = compute_magnitude_angle(loc,
133  ego_vehicle_location,
134  self._vehicle.get_transform().rotation.yaw)
135  if magnitude < 60.0 and angle < min(25.0, min_angle):
136  sel_magnitude = magnitude
137  sel_traffic_light = traffic_light
138  min_angle = angle
139 
140  if sel_traffic_light is not None:
141  if debug:
142  print('=== Magnitude = {} | Angle = {} | ID = {}'.format(
143  sel_magnitude, min_angle, sel_traffic_light.id))
144 
145  if self._last_traffic_light is None:
146  self._last_traffic_light = sel_traffic_light
147

149  return (True, self._last_traffic_light)

151  self._last_traffic_light = None
152

155  def _is_vehicle_hazard(self, vehicle_list):
156  """
157  Check if a given vehicle is an obstacle in our way. To this end we take
158  into account the road and lane the target vehicle is on and run a
159  geometry test to check if the target vehicle is under a certain distance
160  in front of our ego vehicle.
161 
162  WARNING: This method is an approximation that could fail for very large
163  vehicles, which center is actually on a different lane but their
164  extension falls within the ego vehicle lane.
165 
166  :param vehicle_list: list of potential obstacle to check
167  :return: a tuple given by (bool_flag, vehicle), where
168  - bool_flag is True if there is a vehicle ahead blocking us
169  and False otherwise
170  - vehicle is the blocker object itself
171  """
172 
173  ego_vehicle_location = self._vehicle.get_location()
174  ego_vehicle_waypoint = self._map.get_waypoint(ego_vehicle_location)
175 
176  for target_vehicle in vehicle_list:
177  # do not account for the ego vehicle
178  if target_vehicle.id == self._vehicle.id:
179  continue
180 
181  # if the object is not in our lane it's not an obstacle
182  target_vehicle_waypoint = self._map.get_waypoint(target_vehicle.get_location())
183  if target_vehicle_waypoint.road_id != ego_vehicle_waypoint.road_id or \
184  target_vehicle_waypoint.lane_id != ego_vehicle_waypoint.lane_id:
185  continue
186 
187  loc = target_vehicle.get_location()
188  if is_within_distance_ahead(loc, ego_vehicle_location,
189  self._vehicle.get_transform().rotation.yaw,
190  self._proximity_threshold):
191  return (True, target_vehicle)
192 
193  return (False, None)
