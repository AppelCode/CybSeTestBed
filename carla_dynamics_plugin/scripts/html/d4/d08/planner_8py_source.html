134  """

154 def _compute_connection(current_waypoint, next_waypoint):
155 
156  """
157  Compute the type of topological connection between an active waypoint (current_waypoint) and a target waypoint
158  (next_waypoint).
159 
160  :param current_waypoint: active waypoint
161  :param next_waypoint: target waypoint
162  :return: the type of topological connection encoded as a RoadOption enum:
163  RoadOption.STRAIGHT
164  RoadOption.LEFT
165  RoadOption.RIGHT
166  """
167  n = next_waypoint.transform.rotation.yaw
168  n = n % 360.0
169 
170  c = current_waypoint.transform.rotation.yaw
171  c = c % 360.0
172 
173  diff_angle = (n - c) % 180.0
174  if diff_angle < 1.0:
175  return RoadOption.STRAIGHT
176  elif diff_angle > 90.0:
177  return RoadOption.LEFT
178  else:
179  return RoadOption.RIGHT
