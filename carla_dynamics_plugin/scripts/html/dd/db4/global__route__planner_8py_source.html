44  def _build_graph(self):
45  """
46  This function builds a networkx graph representation of topology.
47  The topology is read from self._topology.
48  graph node properties:
49  vertex - (x,y,z) position in world map
50  graph edge properties:
51  entry_vector - unit vector along tangent at entry point
52  exit_vector - unit vector along tangent at exit point
53  net_vector - unit vector of the chord from entry to exit
54  intersection - boolean indicating if the edge belongs to an
55  intersection
56  return : graph -> networkx graph representing the world map,
57  id_map-> mapping from (x,y,z) to node id
58  road_id_to_edge-> map from road id to edge in the graph
59  """
60  graph = nx.DiGraph()
61  id_map = dict() # Map with structure {(x,y,z): id, ... }
62  road_id_to_edge = dict() # Map with structure {road_id: {lane_id: edge, ... }, ... }
63 
64  for segment in self._topology:
65 
66  entry_xyz, exit_xyz = segment['entryxyz'], segment['exitxyz']
67  path = segment['path']
68  entry_wp, exit_wp = segment['entry'], segment['exit']
69  intersection = entry_wp.is_intersection
70  road_id, lane_id = entry_wp.road_id, entry_wp.lane_id
71 
72  for vertex in entry_xyz, exit_xyz:
73  # Adding unique nodes and populating id_map
74  if vertex not in id_map:
75  new_id = len(id_map)
76  id_map[vertex] = new_id
77  graph.add_node(new_id, vertex=vertex)
78  n1 = id_map[entry_xyz]
79  n2 = id_map[exit_xyz]
80  if road_id not in road_id_to_edge:
81  road_id_to_edge[road_id] = dict()
82  road_id_to_edge[road_id][lane_id] = (n1, n2)
83 
84  # Adding edge with attributes
85  graph.add_edge(
86  n1, n2,
87  length=len(path) + 1, path=path,
88  entry_waypoint=entry_wp, exit_waypoint=exit_wp,
89  entry_vector=vector(
90  entry_wp.transform.location,
91  path[0].transform.location if len(path) > 0 else exit_wp.transform.location),
92  exit_vector=vector(
93  path[-1].transform.location if len(path) > 0 else entry_wp.transform.location,
94  exit_wp.transform.location),
95  net_vector=vector(entry_wp.transform.location, exit_wp.transform.location),
96  intersection=intersection, type=RoadOption.LANEFOLLOW)
97 
98  return graph, id_map, road_id_to_edge

162  l1 = np.array(self._graph.nodes[n1]['vertex'])

166  def _path_search(self, origin, destination):
167  """
168  This function finds the shortest path connecting origin and destination
169  using A* search with distance heuristic.
170  origin : carla.Location object of start position
171  destination : carla.Location object of of end position
172  return : path as list of node ids (as int) of the graph self._graph
173  connecting origin and destination
174  """
175 
176  start, end = self._localize(origin), self._localize(destination)
177 
178  route = nx.astar_path(
179  self._graph, source=start[0], target=end[0],
180  heuristic=self._distance_heuristic, weight='length')
181  route.append(end[1])
182  return route

184  def _turn_decision(self, index, route, threshold=math.radians(5)):
185  """
186  This method returns the turn decision (RoadOption) for pair of edges
187  around current index of route list
188  """
189 
190  decision = None
191  previous_node = route[index-1]
192  current_node = route[index]
193  next_node = route[index+1]
194  next_edge = self._graph.edges[current_node, next_node]
195  if index > 0:
196  current_edge = self._graph.edges[previous_node, current_node]
197  calculate_turn = current_edge['type'].value == RoadOption.LANEFOLLOW.value and \
198  not current_edge['intersection'] and \
199  next_edge['type'].value == RoadOption.LANEFOLLOW.value and \
200  next_edge['intersection']
201  if calculate_turn:
202  cv, nv = current_edge['exit_vector'], next_edge['net_vector']
203  cross_list = []
204  for neighbor in self._graph.successors(current_node):
205  select_edge = self._graph.edges[current_node, neighbor]
206  if select_edge['type'].value == RoadOption.LANEFOLLOW.value:
207  if neighbor != route[index+1]:
208  sv = select_edge['net_vector']
209  cross_list.append(np.cross(cv, sv)[2])
210  next_cross = np.cross(cv, nv)[2]
211  deviation = math.acos(np.dot(cv, nv) /\
212  (np.linalg.norm(cv)*np.linalg.norm(nv)))
213  if not cross_list:
214  cross_list.append(0)
215  if deviation < threshold:
216  decision = RoadOption.STRAIGHT
217  elif cross_list and next_cross < min(cross_list):
218  decision = RoadOption.LEFT
219  elif cross_list and next_cross > max(cross_list):
220  decision = RoadOption.RIGHT
221  else:
222  decision = next_edge['type']
223  else:
224  decision = next_edge['type']
225 
226  return decision

260  def trace_route(self, origin, destination):
261  """
262  This method returns list of (carla.Waypoint, RoadOption) from origin to destination
263  """
264 
265  route_trace = []
266  route = self._path_search(origin, destination)
267  current_waypoint = self._dao.get_waypoint(origin)
268  resolution = self._dao.get_resolution()
269 
270  for i in range(len(route) - 1):
271  road_option = self._turn_decision(i, route)
272  edge = self._graph.edges[route[i], route[i+1]]
273  path = []
274 
275  if edge['type'].value != RoadOption.LANEFOLLOW.value and edge['type'].value != RoadOption.VOID.value:
276  n1, n2 = self._road_id_to_edge[edge['exit_waypoint'].road_id][edge['exit_waypoint'].lane_id]
277  next_edge = self._graph.edges[n1, n2]
278  if next_edge['path']:
279  closest_index = self._find_closest_in_list(current_waypoint, next_edge['path'])
280  closest_index = min(len(next_edge['path'])-1, closest_index+5)
281  current_waypoint = next_edge['path'][closest_index]
282  else:
283  current_waypoint = next_edge['exit_waypoint']
284  route_trace.append((current_waypoint, road_option))
285 
286  else:
287  path = path + [edge['entry_waypoint']] + edge['path'] + [edge['exit_waypoint']]
288  closest_index = self._find_closest_in_list(current_waypoint, path)
289  for waypoint in path[closest_index:]:
290  current_waypoint = waypoint
291  route_trace.append((current_waypoint, road_option))
292  if len(route)-i <= 2 and waypoint.transform.location.distance(destination) < 2*resolution:
293  break
