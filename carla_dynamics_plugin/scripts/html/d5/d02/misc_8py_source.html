18 def draw_waypoints(world, waypoints, z=0.5):

20  Draw a list of waypoints at a certain height given in z.

25  :return:
26  """

45 def is_within_distance_ahead(target_location, current_location, orientation, max_distance):
46  """
47  Check if a target object is within a certain distance in front of a reference object.
48 
49  :param target_location: location of the target object
50  :param current_location: location of the reference object
51  :param orientation: orientation of the reference object
52  :param max_distance: maximum allowed distance
53  :return: True if target object is within max_distance ahead of the reference object
54  """
55  target_vector = np.array([target_location.x - current_location.x, target_location.y - current_location.y])
56  norm_target = np.linalg.norm(target_vector)
57 
58  # If the vector is too short, we can simply stop here
59  if norm_target < 0.001:
60  return True
61 
62  if norm_target > max_distance:
63  return False
64 
65  forward_vector = np.array(
66  [math.cos(math.radians(orientation)), math.sin(math.radians(orientation))])
67  d_angle = math.degrees(math.acos(np.dot(forward_vector, target_vector) / norm_target))
68 
69  return d_angle < 90.0

72 def compute_magnitude_angle(target_location, current_location, orientation):
73  """
74  Compute relative angle and distance between a target_location and a current_location
75 
76  :param target_location: location of the target object
77  :param current_location: location of the reference object
78  :param orientation: orientation of the reference object
79  :return: a tuple composed by the distance to the object and the angle between both objects
80  """
81  target_vector = np.array([target_location.x - current_location.x, target_location.y - current_location.y])
82  norm_target = np.linalg.norm(target_vector)
83 
84  forward_vector = np.array([math.cos(math.radians(orientation)), math.sin(math.radians(orientation))])
85  d_angle = math.degrees(math.acos(np.dot(forward_vector, target_vector) / norm_target))

98 def vector(location_1, location_2):
99  """
100  Returns the unit vector from location_1 to location_2
101  location_1, location_2: carla.Location objects
102  """
103  x = location_2.x - location_1.x

105  z = location_2.z - location_1.z
106  norm = np.linalg.norm([x, y, z]) + np.finfo(float).eps

108  return [x / norm, y / norm, z / norm]
