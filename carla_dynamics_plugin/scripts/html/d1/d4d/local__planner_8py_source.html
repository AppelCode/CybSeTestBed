33 class LocalPlanner(object):
34  """
35  LocalPlanner implements the basic behavior of following a trajectory of waypoints that is generated on-the-fly.
36  The low-level motion of the vehicle is computed by using two PID controllers, one is used for the lateral control
37  and the other for the longitudinal control (cruise speed).
38 
39  When multiple paths are available (intersections) this local planner makes a random choice.
40  """

153  def _compute_next_waypoints(self, k=1):
154  """
155  Add new waypoints to the trajectory queue.
156 
157  :param k: how many waypoints to compute
158  :return:
159  """
160  # check we do not overflow the queue
161  available_entries = self._waypoints_queue.maxlen - len(self._waypoints_queue)
162  k = min(available_entries, k)
163 
164  for _ in range(k):
165  last_waypoint = self._waypoints_queue[-1][0]
166  next_waypoints = list(last_waypoint.next(self._sampling_radius))
167 
168  if len(next_waypoints) == 1:
169  # only one option available ==> lanefollowing
170  next_waypoint = next_waypoints[0]
171  road_option = RoadOption.LANEFOLLOW
172  else:
173  # random choice between the possible options
174  road_options_list = _retrieve_options(
175  next_waypoints, last_waypoint)
176  road_option = random.choice(road_options_list)
177  next_waypoint = next_waypoints[road_options_list.index(
178  road_option)]
179 
180  self._waypoints_queue.append((next_waypoint, road_option))

189  def run_step(self, debug=False):
190  """
191  Execute one step of local planning which involves running the longitudinal and lateral PID controllers to
192  follow the waypoints trajectory.
193 
194  :param debug: boolean flag to activate waypoints debugging
195  :return:
196  """
197 
198  # not enough waypoints in the horizon? => add more!
199  if not self._global_plan and len(self._waypoints_queue) < int(self._waypoints_queue.maxlen * 0.5):
200  self._compute_next_waypoints(k=100)
201 
202  if len(self._waypoints_queue) == 0:
203  control = carla.VehicleControl()
204  control.steer = 0.0
205  control.throttle = 0.0
206  control.brake = 1.0
207  control.hand_brake = False
208  control.manual_gear_shift = False
209 
210  return control
211 
212  # Buffering the waypoints
213  if not self._waypoint_buffer:
214  for i in range(self._buffer_size):
215  if self._waypoints_queue:
216  self._waypoint_buffer.append(
217  self._waypoints_queue.popleft())
218  else:
219  break
220 
221  # current vehicle waypoint
222  self._current_waypoint = self._map.get_waypoint(self._vehicle.get_location())
223  # target waypoint
224  self.target_waypoint, self._target_road_option = self._waypoint_buffer[0]
225  # move using PID controllers
226  control = self._vehicle_controller.run_step(self._target_speed, self.target_waypoint)
227 
228  # purge the queue of obsolete waypoints
229  vehicle_transform = self._vehicle.get_transform()
230  max_index = -1
231 
232  for i, (waypoint, _) in enumerate(self._waypoint_buffer):
233  if distance_vehicle(
234  waypoint, vehicle_transform) < self._min_distance:
235  max_index = i
236  if max_index >= 0:
237  for i in range(max_index + 1):
238  self._waypoint_buffer.popleft()

246 def _retrieve_options(list_waypoints, current_waypoint):
247  """
248  Compute the type of connection between the current active waypoint and the multiple waypoints present in
249  list_waypoints. The result is encoded as a list of RoadOption enums.
250 
251  :param list_waypoints: list with the possible target waypoints in case of multiple options
252  :param current_waypoint: current active waypoint
253  :return: list of RoadOption enums representing the type of connection from the active waypoint to each
254  candidate in list_waypoints
255  """
256  options = []
257  for next_waypoint in list_waypoints:
258  # this is needed because something we are linking to
259  # the beggining of an intersection, therefore the
260  # variation in angle is small
261  next_next_waypoint = next_waypoint.next(3.0)[0]
262  link = _compute_connection(current_waypoint, next_next_waypoint)
263  options.append(link)
264 
265  return options

268 def _compute_connection(current_waypoint, next_waypoint):
269  """
270  Compute the type of topological connection between an active waypoint (current_waypoint) and a target waypoint
271  (next_waypoint).
272 
273  :param current_waypoint: active waypoint
274  :param next_waypoint: target waypoint
275  :return: the type of topological connection encoded as a RoadOption enum:
276  RoadOption.STRAIGHT
277  RoadOption.LEFT
278  RoadOption.RIGHT
279  """
280  n = next_waypoint.transform.rotation.yaw
281  n = n % 360.0
282 
283  c = current_waypoint.transform.rotation.yaw
284  c = c % 360.0
285 
286  diff_angle = (n - c) % 180.0
287  if diff_angle < 1.0:
288  return RoadOption.STRAIGHT
289  elif diff_angle > 90.0:
290  return RoadOption.LEFT
291  else:
292  return RoadOption.RIGHT
